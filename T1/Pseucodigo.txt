	LEER nombre
	ESCRIBIR "Dime el tipo de trabajador de : " nombre

	MIENTRAS (tipo <> 'A' || tipo <> 'B' || tipo <> 'C')
		ESCRIBIR: "Valor incorrecto, introduzca A, B o C."
